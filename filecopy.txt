#include <stdio.h>
#include <fcntl.h>
#include <sys/types.h>
#include <unistd.h>

int main(int argc, char *argv[])
{

  char buff[128];      //used as a buffer to read a file in 128 Byte chunks for the copy
  char currentChar[1]; //used as temporary storage when counting the number of characters in the input file
  int numChars = 0;    //used to count the number of characters in the input file
  int i;               //used to iterate through the copying process after finding the number of character to traverse

  //open the first file
  int f1check = open(argv[1], O_RDONLY);
  if (f1check < 0) {
    printf("Failed to open file 1\n"); //report an error message if the open was unsuccessful
    return 1;
  } else {
    printf("Success for file 1\n");
  }

  //open the second file
  int f2check = open(argv[2], O_WRONLY);
  if (f2check < 0) {
    printf("Failed to open file 2\n"); //report an error message if the open was unsuccessful
    return -1;
  } else {
    printf("Success for file 2\n");
  }

  //count the number of characters in the input file
  while (read(f1check, currentChar, 1) != '\0') {
    numChars++;
  }

  //close the file and report the number of characters
  //this is to reset the postion for the read to the beginning of the file again
  close(f1check);
  printf("The number of characters in the file is: %d\n", numChars);
  
  //open the first file again. This time for copying
  f1check = open (argv[1], O_RDONLY);
  if (f1check < 0)
    {
      printf("Failed to open file 1 the second time.\n"); //report an error message if the open was unsuccessful
      return -1;
    } else {
    printf("Successfully opened file 1 the second time.");
  }

  //loop through the file reading 128 Byte chunks and writing them to the output file.
  for (i = 0; i < numChars/128; i++)
    {
      //read a chunk of data from the input file
      if (read(f1check, buff, 128) < 0) {
	printf("Error reading file1\n"); //report an error message if the read was unsuccessful
	return -1;
      }else {
	printf("Read successful\n");
      }
      //then write that chunk to the output file
      if (write(f2check, buff, 128) < 0) {
	printf("Error writing to file2\n"); //report an error message if the write was unsuccessful
	return -1;
      } else {
	printf("Write successful\n");
      }
    }

  //the previous loop will leave some characters uncopied if the length of the file is not
  //a multiple of 256 Bytes
  //to deal with any remaining characters we execute the following:
  //first, read a chunk
  if (read(f1check, buff, numChars % 128) < 0) {
    printf("Error reading the end of file 1\n"); //report an error message if the read was unsuccessful
    return -1;
  } else {
    printf("success reading end of file 1!\n");
  }
  //then, write that chunk to the ouput file
  if (write(f2check, buff, numChars % 128) < 0) {
    printf("error writing to the end of file 2.\n"); //report an error message if the write was unsuccessful
    return -1;
  } else {
    printf("Success writing to the end of file 2\n");
  }

  //finally, close both files
  close(f1check);
  close(f2check);
  
  return 0;
}
